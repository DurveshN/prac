<h2>PHP Array Program</h2>
<?php
// 1. Indexed Array
$names = array("Amit", "Neha", "Rahul", "Riya");
// Display Indexed Array
echo "<h3>Indexed Array:</h3>";
foreach ($names as $name) {
echo $name . "<br>";
}
// Count elements
echo "Total Names: " . count($names) . "<br><br>";
// 2. Associative Array
$marks = array(
"Amit" => 78,
"Neha" => 85,
"Rahul" => 82
);
// Display Associative Array
echo "<h3>Associative Array:</h3>";
foreach ($marks as $name => $mark) {
echo $name . " : " . $mark . "<br>";
}
echo "<br>"; // Added line break for consistent formatting
// Count elements
echo "Total Students: " . count($marks);
?>
